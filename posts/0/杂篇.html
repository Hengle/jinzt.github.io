注册监听前后台的切换事件

onLoad: function () {
    cc.game.on(cc.game.EVENT_HIDE, function(){
    console.log("游戏进入后台");
    this.onHideGame();//处理游戏切到后台时的事件
    },this);

    cc.game.on(cc.game.EVENT_SHOW, function(){
    console.log("重新返回游戏");
    this.onShowGame();//处理游戏切回前台时的事件
    },this);
},
